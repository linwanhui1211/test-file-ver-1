<script>

    const vocabularyList = [

        { word: "因素", pinyin: "yīn sù", english: "factor", meaning: "决定事物成败的原因或条件。", sentence: "努力是取得成功的关键____。" },

        { word: "水喉", pinyin: "shuǐ hóu", english: "water tap", meaning: "自来水管的开关。", sentence: "洗手后，请记得关紧____。" },

        { word: "资源", pinyin: "zī yuán", english: "resources", meaning: "生产或生活资料的天然来源。", sentence: "水是宝贵的____，我们应该珍惜。" },

        { word: "艰巨", pinyin: "jiān jù", english: "arduous", meaning: "困难而繁重。", sentence: "这是一项____的任务，需要团队合作。" },

        { word: "昂贵", pinyin: "áng guì", english: "expensive", meaning: "价格很高。", sentence: "这台净水机器的维护费用非常____。" },

        { word: "消耗", pinyin: "xiāo hào", english: "consume", meaning: "因使用而逐渐减少。", sentence: "洗澡时间过长会____大量的水。" },

        { word: "干旱", pinyin: "gān hàn", english: "drought", meaning: "长期无雨造成的缺水现象。", sentence: "由于长期____，土地都裂开了。" },

        { word: "扩大", pinyin: "kuò dà", english: "expand", meaning: "使范围、规模比原来大。", sentence: "新加坡正在____集水区的范围。" },

        { word: "蓄水池", pinyin: "xù shuǐ chí", english: "reservoir", meaning: "人工修建的储水设施。", sentence: "麦里芝____风景优美，还能储水。" },

        { word: "洗澡", pinyin: "xǐ zǎo", english: "take a shower", meaning: "用水洗去身上的污垢。", sentence: "节水小卫士____时会缩短时间。" },

        { word: "浴室", pinyin: "yù shì", english: "bathroom", meaning: "专门用来洗澡的房间。", sentence: "离开____前，请确保水龙头关好。" },

        { word: "趋势", pinyin: "qū shì", english: "trend", meaning: "事物发展的动向。", sentence: "全球气候变暖已经成为一种____。" },

        { word: "开源节流", pinyin: "kāi yuán jié liú", english: "tap new sources and reduce usage", meaning: "开发水源，节约水流。", sentence: "我们要做到____，才能确保水供充足。" }

    ];

 

    let score = 0;

    let isPlaying = false;

    let currentVocab = null;

    let items = [];

    let fallSpeed = 2.5;

    let animationId;

    let spawnId;

 

    const playArea = document.getElementById('play-area');

    const scoreText = document.getElementById('score');

    const promptArea = document.getElementById('prompt-area');

    const toast = document.getElementById('toast');

 

    function showHint(type) {

        if (!isPlaying || !currentVocab) return;

        toast.innerText = type === 'pinyin' ? `拼音：${currentVocab.pinyin}` : `英文：${currentVocab.english}`;

        toast.style.opacity = 1;

        setTimeout(() => toast.style.opacity = 0, 2000);

    }

 

    function playAudio() {

        if (!isPlaying || !currentVocab || !('speechSynthesis' in window)) return;

        const text = promptArea.innerText.replace(/【.*?】/, '').replace('____', currentVocab.word);

        const utter = new SpeechSynthesisUtterance(text);

        utter.lang = 'zh-CN';

        window.speechSynthesis.speak(utter);

    }

 

    function generateQuestion() {

        currentVocab = vocabularyList[Math.floor(Math.random() * vocabularyList.length)];

        const type = Math.floor(Math.random() * 2);

        promptArea.innerText = type === 0 ? `【填空】${currentVocab.sentence}` : `【意思】${currentVocab.meaning}`;

    }

 

    function spawnItem() {

        if (!isPlaying) return;

        const item = document.createElement('div');

        item.className = 'falling-item';

        const typeRand = Math.random();

        let type, content;

 

        if (typeRand < 0.6) {

            type = 'raindrop';

            item.classList.add('raindrop');

            content = Math.random() > 0.4 ? currentVocab.word : vocabularyList[Math.floor(Math.random() * vocabularyList.length)].word;

            item.innerText = content;

        } else if (typeRand < 0.8) {

            type = 'bomb';

            item.classList.add('bomb');

            item.innerText = '💣';

        } else {

            type = 'flower';

            item.classList.add('flower');

            item.innerText = '🌸';

        }

 

        const x = Math.random() * (window.innerWidth - 80) + 10;

        item.style.left = `${x}px`;

        let y = -100;

        item.style.top = `${y}px`;

 

        const clickHandler = (e) => {

            e.preventDefault();

            if (!isPlaying) return;

            handleImpact(type, content, item, x, y);

        };

        item.addEventListener('touchstart', clickHandler);

        item.addEventListener('mousedown', clickHandler);

 

        playArea.appendChild(item);

        items.push({ el: item, y, type, content, x });

 

        spawnId = setTimeout(spawnItem, Math.max(700, 1800 - score * 5));

    }

 

    function handleImpact(type, content, el, x, y) {

        let change = 0;

        const currentY = parseFloat(el.style.top);

        if (type === 'raindrop') {

            if (content === currentVocab.word) {

                change = 10;

                generateQuestion();

                fallSpeed += 0.1;

            } else {

                change = -5;

            }

        } else if (type === 'bomb') {

            change = -10;

        } else {

            change = 5;

        }

 

        score += change;

        scoreText.innerText = score;

        showPopup(change, x, currentY);

        el.remove();

        items = items.filter(i => i.el !== el);

        if (score <= -30) endGame();

    }

 

    function showPopup(num, x, y) {

        const p = document.createElement('div');

        p.className = `score-popup ${num > 0 ? 'good' : 'bad'}`;

        p.innerText = num > 0 ? `+${num}` : num;

        p.style.left = `${x}px`;

        p.style.top = `${y}px`;

        playArea.appendChild(p);

        setTimeout(() => p.remove(), 800);

    }

 

    function gameLoop() {

        if (!isPlaying) return;

        items.forEach((item, index) => {

            item.y += fallSpeed;

            item.el.style.top = `${item.y}px`;

            if (item.y > window.innerHeight) {

                item.el.remove();

                items.splice(index, 1);

            }

        });

        animationId = requestAnimationFrame(gameLoop);

    }

 

    function startGame() {

        score = 0;

        fallSpeed = 2.5;

        items = [];

        isPlaying = true;

        scoreText.innerText = '0';

        playArea.innerHTML = '';

        document.getElementById('start-screen').style.display = 'none';

        document.getElementById('game-over').style.display = 'none';

        generateQuestion();

        spawnItem();

        gameLoop();

    }

 

    function endGame() {

        isPlaying = false;

        clearTimeout(spawnId);

        cancelAnimationFrame(animationId);

        document.getElementById('final-score').innerText = score;

        document.getElementById('game-over').style.display = 'flex';

    }

</script>
