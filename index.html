<script>

    // 词汇库 (包含了拼音、意思、例句、英文)

    const vocabularyList = [

        { word: "因素", pinyin: "yīn sù", english: "factor / element", meaning: "决定事物成败的原因或条件。", sentence: "努力是取得成功的关键____。" },

        { word: "水喉", pinyin: "shuǐ hóu", english: "water tap", meaning: "自来水管的开关。", sentence: "洗手后，请记得关紧____，不要浪费水。" },

        { word: "资源", pinyin: "zī yuán", english: "resources", meaning: "生产资料或生活资料的天然来源。", sentence: "水是宝贵的____，我们应该珍惜。" },

        { word: "艰巨", pinyin: "jiān jù", english: "arduous / formidable", meaning: "困难而繁重。", sentence: "这是一项____的任务，我们需要团队合作。" },

        { word: "昂贵", pinyin: "áng guì", english: "expensive", meaning: "价格很高。", sentence: "生产新生水的机器十分____。" },

        { word: "消耗", pinyin: "xiāo hào", english: "consume / deplete", meaning: "因使用或受损失而逐渐减少。", sentence: "跑马拉松____了他大量的体力。" },

        { word: "干旱", pinyin: "gān hàn", english: "drought / dry", meaning: "因长期无雨或少雨而造成土壤缺水。", sentence: "由于长时间不下雨，这片土地变得非常____。" },

        { word: "扩大", pinyin: "kuò dà", english: "expand / enlarge", meaning: "使范围、规模等比原来大。", sentence: "学校计划____图书馆的面积。" },

        { word: "蓄水池", pinyin: "xù shuǐ chí", english: "reservoir", meaning: "用人工材料修建的储存水的建筑物。", sentence: "麦里芝____是新加坡著名的自然保护区之一。" },

        { word: "洗澡", pinyin: "xǐ zǎo", english: "take a bath / shower", meaning: "用水洗去身上的污垢。", sentence: "运动流满身汗后，去____会感觉很舒服。" },

        { word: "浴室", pinyin: "yù shì", english: "bathroom", meaning: "专门用来洗澡的房间。", sentence: "他在____里一边淋浴一边唱歌。" },

        { word: "趋势", pinyin: "qū shì", english: "trend", meaning: "事物发展的动向。", sentence: "随着环保意识的提高，节约用水已经成为一种____。" },

        { word: "开源节流", pinyin: "kāi yuán jié liú", english: "tap new sources and reduce usage", meaning: "开发水源，节约水流。", sentence: "为了应对水荒，我们必须做到____。" }

    ];

 

    // 游戏状态变量

    let score = 0;

    let isPlaying = false;

    let currentVocab = null;

    let currentPromptText = "";

    let fallSpeed = 2; // 初始掉落速度

    let animationFrameId;

    let spawnTimeoutId;

    let items = []; // 存储屏幕上的所有掉落物

 

    // DOM 元素引用

    const playArea = document.getElementById('play-area');

    const scoreDisplay = document.getElementById('score');

    const promptArea = document.getElementById('prompt-area');

    const toastEl = document.getElementById('toast');

 

    // 随机获取题目格式（词意、例句 或 拼音）

    function generateQuestion() {

        const randomIndex = Math.floor(Math.random() * vocabularyList.length);

        currentVocab = vocabularyList[randomIndex];

       

        // 随机决定提示形式：0=例句，1=意思，2=拼音

        const promptType = Math.floor(Math.random() * 3);

       

        if (promptType === 0) {

            currentPromptText = "【填空】" + currentVocab.sentence;

        } else if (promptType === 1) {

            currentPromptText = "【意思】" + currentVocab.meaning;

        } else {

            currentPromptText = "【拼音】" + currentVocab.pinyin;

        }

       

        promptArea.innerHTML = currentPromptText;

    }

 

    // 显示游戏内提示 (Toast)

    function showToast(message) {

        toastEl.innerText = message;

        toastEl.style.opacity = 1;

        setTimeout(() => { toastEl.style.opacity = 0; }, 2000);

    }

 

    // 辅助工具按钮功能

    function showHint(type) {

        if (!isPlaying || !currentVocab) return;

        if (type === 'pinyin') {

            showToast("拼音： " + currentVocab.pinyin);

        } else if (type === 'english') {

            showToast("英文： " + currentVocab.english);

        }

    }

 

    // 朗读功能 (TTS)

    function playAudio() {

        if (!isPlaying || !currentVocab) return;

        if ('speechSynthesis' in window) {

            // 提取要读的纯文本，把____替换为“什么”

            let textToSpeak = currentPromptText.replace(/【.*?】/, '').replace('____', '什么');

            const utterance = new SpeechSynthesisUtterance(textToSpeak);

            utterance.lang = 'zh-CN'; // 设定为中文

            utterance.rate = 0.9;     // 语速稍微放慢方便聆听

            window.speechSynthesis.speak(utterance);

        } else {

            showToast("您的浏览器不支持朗读功能哦！");

        }

    }

 

    // 创建掉落物

    function spawnItem() {

        if (!isPlaying) return;

 

        const itemEl = document.createElement('div');

        itemEl.className = 'falling-item';

       

        // 决定掉落物类型：60%水滴(答题)，20%炸弹，20%花朵

        const rand = Math.random();

        let itemType, itemData;

 

        if (rand < 0.6) {

            itemType = 'raindrop';

            itemEl.classList.add('raindrop');

            // 一半几率是正确答案，一半是干扰词

            if (Math.random() > 0.5) {

                itemData = currentVocab.word;

            } else {

                const randomWrongVocab = vocabularyList[Math.floor(Math.random() * vocabularyList.length)];

                itemData = randomWrongVocab.word;

            }

            // 适配较长词汇的字体大小

            if (itemData.length > 2) {

                itemEl.style.fontSize = '14px';

            }

            itemEl.innerText = itemData;

        } else if (rand < 0.8) {

            itemType = 'bomb';

            itemEl.classList.add('bomb');

            itemEl.innerText = '💣';

        } else {

            itemType = 'flower';

            itemEl.classList.add('flower');

            itemEl.innerText = '🌸';

        }

 

        // 随机水平位置，留出安全边距避免溢出

        const safeWidth = window.innerWidth - 80;

        const xPos = Math.max(10, Math.random() * safeWidth);

       

        itemEl.style.left = xPos + 'px';

        let yPos = -100; // 从屏幕上方外开始

        itemEl.style.top = yPos + 'px';

 

        // 绑定点击/触摸事件

        const handleInteraction = (e) => {

            e.preventDefault(); // 防止双击放大等默认行为

            if (!isPlaying) return;

            checkAnswer(itemType, itemData, itemEl, xPos, yPos);

        };

       

        itemEl.addEventListener('mousedown', handleInteraction);

        itemEl.addEventListener('touchstart', handleInteraction, {passive: false});

 

        playArea.appendChild(itemEl);

        items.push({ el: itemEl, y: yPos, type: itemType, text: itemData });

 

        // 随着分数增加，产生物品的速度加快

        let nextSpawnTime = Math.max(800, 2000 - (score * 5));

        spawnTimeoutId = setTimeout(spawnItem, nextSpawnTime);

    }

 

    // 验证点击结果并计分

    function checkAnswer(type, text, el, x, y) {

        let scoreChange = 0;

       

        // 获取元素当前准确的Y坐标以显示反馈

        const currentY = parseFloat(el.style.top);

 

        if (type === 'raindrop') {

            if (text === currentVocab.word) {

                scoreChange = 10;

                playEffectSound('correct');

                generateQuestion(); // 换下一题

                fallSpeed = Math.min(8, fallSpeed + 0.1); // 答对加速，增加难度

            } else {

                scoreChange = -5;

                playEffectSound('wrong');

            }

        } else if (type === 'bomb') {

            scoreChange = -10;

            playEffectSound('bomb');

        } else if (type === 'flower') {

            scoreChange = 5;

            playEffectSound('bonus');

        }

 

        updateScore(scoreChange, x, currentY);

 

        // 移除被点击的物品

        el.remove();

        items = items.filter(i => i.el !== el);

    }

 

    // 更新分数并显示浮动动画

    function updateScore(change, x, y) {

        score += change;

        scoreDisplay.innerText = score;

       

        // 创建浮动分数反馈

        const feedback = document.createElement('div');

        feedback.className = 'score-popup ' + (change > 0 ? 'good' : 'bad');

        feedback.innerText = (change > 0 ? '+' : '') + change;

        feedback.style.left = (x + 20) + 'px';

        feedback.style.top = y + 'px';

       

        playArea.appendChild(feedback);

        setTimeout(() => feedback.remove(), 800);

 

        // 游戏结束条件判断 (例如低于 -20 分)

        if (score <= -20) {

            endGame();

        }

    }

 

    // 简单的网页音效模拟 (使用 AudioContext)

    const audioCtx = new (window.AudioContext || window.webkitAudioContext)();

    function playEffectSound(type) {

        if (audioCtx.state === 'suspended') audioCtx.resume();

        const oscillator = audioCtx.createOscillator();

        const gainNode = audioCtx.createGain();

        oscillator.connect(gainNode);

        gainNode.connect(audioCtx.destination);

       

        if (type === 'correct') {

            oscillator.type = 'sine';

            oscillator.frequency.setValueAtTime(600, audioCtx.currentTime);

            oscillator.frequency.exponentialRampToValueAtTime(1200, audioCtx.currentTime + 0.1);

            gainNode.gain.setValueAtTime(0.5, audioCtx.currentTime);

            gainNode.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.2);

            oscillator.start(); oscillator.stop(audioCtx.currentTime + 0.2);

        } else if (type === 'wrong') {

            oscillator.type = 'sawtooth';

            oscillator.frequency.setValueAtTime(300, audioCtx.currentTime);

            oscillator.frequency.linearRampToValueAtTime(150, audioCtx.currentTime + 0.2);

            gainNode.gain.setValueAtTime(0.5, audioCtx.currentTime);

            gainNode.gain.linearRampToValueAtTime(0.01, audioCtx.currentTime + 0.3);

            oscillator.start(); oscillator.stop(audioCtx.currentTime + 0.3);

        } else if (type === 'bonus') {

            oscillator.type = 'square';

            oscillator.frequency.setValueAtTime(400, audioCtx.currentTime);

            oscillator.frequency.setValueAtTime(800, audioCtx.currentTime + 0.1);

            gainNode.gain.setValueAtTime(0.3, audioCtx.currentTime);

            gainNode.gain.linearRampToValueAtTime(0.01, audioCtx.currentTime + 0.3);

            oscillator.start(); oscillator.stop(audioCtx.currentTime + 0.3);

        } else if (type === 'bomb') {

            oscillator.type = 'square';

            oscillator.frequency.setValueAtTime(100, audioCtx.currentTime);

            gainNode.gain.setValueAtTime(0.8, audioCtx.currentTime);

            gainNode.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.5);

            oscillator.start(); oscillator.stop(audioCtx.currentTime + 0.5);

        }

    }

 

    // 游戏主动画循环

    function gameLoop() {

        if (!isPlaying) return;

       

        const screenHeight = window.innerHeight;

 

        for (let i = items.length - 1; i >= 0; i--) {

            let item = items[i];

            item.y += fallSpeed; // 控制掉落

            item.el.style.top = item.y + 'px';

 

            // 如果物品掉出屏幕底部，则移除

            if (item.y > screenHeight) {

                item.el.remove();

                items.splice(i, 1);

            }

        }

       

        animationFrameId = requestAnimationFrame(gameLoop);

    }

 

    // 初始化并开始游戏

    function startGame() {

        document.getElementById('start-screen').style.display = 'none';

        document.getElementById('game-over').style.display = 'none';

       

        // 唤醒音频引擎

        if (audioCtx.state === 'suspended') audioCtx.resume();

       

        // 清理旧数据

        playArea.innerHTML = '';

        items = [];

        score = 0;

        fallSpeed = 2.5; // 难度重置

        scoreDisplay.innerText = score;

       

        isPlaying = true;

        generateQuestion();

       

        clearTimeout(spawnTimeoutId);

        cancelAnimationFrame(animationFrameId);

       

        spawnItem();

        gameLoop();

    }

 

    // 结束游戏

    function endGame() {

        isPlaying = false;

        clearTimeout(spawnTimeoutId);

        cancelAnimationFrame(animationFrameId);

        document.getElementById('final-score').innerText = score;

        document.getElementById('game-over').style.display = 'flex';

    }
